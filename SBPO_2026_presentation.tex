\section{Motivation \& Context}

\begin{frame}[plain]
\begin{center}
{\Huge Motivation \\ \& \\ Context}
\end{center}
\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Scheduling FOTA Updates for Connected Vehicles}

\textbf{Firmware-Over-The-Air (FOTA)} updates must be delivered to
\textbf{millions of connected vehicles} through a cellular network.

~\\

Key challenges:

\begin{itemize}
\item Vehicles are only \textbf{available in certain time periods} and \textbf{cell sectors}.

~\\

\item Each cell sector has \textbf{time-varying capacity}.

~\\

\item A global \textbf{bandwidth cap} limits total throughput per period.

~\\

\item The scheduling must be done at \textbf{large scale} (up to 10,000 jobs or machines).
\end{itemize}

\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{From TMDSP to MOTMDSP}

The \textbf{Time- and Machine-Dependent Scheduling Problem (TMDSP)} minimizes only
$\sum_j C_j$ (total completion time).

~\\

\textbf{Problem:} minimizing $\sum_j C_j$ does \emph{not} necessarily minimize $C_{\max}$.

~\\

\begin{itemize}
\item The correlation between $\sum_j C_j$ and $C_{\max}$ is \textbf{weak}.

~\\

\item An optimal $\sum_j C_j$ schedule may leave some jobs finishing \textbf{much later} than others.
\end{itemize}

~\\

$\Rightarrow$ We propose the \textbf{Multi-Objective TMDSP (MOTMDSP)}: minimize both
$C_{\max}$ and $\sum_j C_j$ \textbf{simultaneously}.

\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Multi-Objective Trade-Off}

\begin{center}

\section{Problem Definition}

\begin{frame}[plain]
\begin{center}
{\Huge Problem \\ Definition}
\end{center}
\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{MOTMDSP -- Key Elements}

\begin{columns}[c,onlytextwidth]
\column{0.52\textwidth}
\textbf{Sets:}
\begin{itemize}
\item $J$: jobs (vehicles to update)
\item $M$: machines (cell sectors)
\item $T$: discrete time periods
\end{itemize}

~\\

\textbf{Parameters:}
\begin{itemize}
\item $c_{mt}$: capacity of machine $m$ at period $t$
\item $\eta_{mt}$: production rate (data per unit)
\item $q_{jmt}$: resource consumption of job $j$
\item $d_j$: total demand of job $j$
\end{itemize}

\column{0.48\textwidth}
\textbf{Constraints:}
\begin{itemize}
\item Each job assigned to $\le 1$ machine per period
\item Machine capacity $c_{mt}$ respected
\item Global bandwidth cap $r_{\max}$
\item At most $k_{\max}$ active jobs per period
\item Minimum resource usage $r_{\min}$
\end{itemize}

~\\

\textbf{Decision:} assign each job to machines and periods until its demand $d_j$ is fully served.
\end{columns}

\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Objective Functions}

\textbf{Minimize simultaneously:}

~\\

\begin{columns}[c,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Makespan}
\[
\min \; C_{\max} = \max_{j \in J} C_j
\]
The \textbf{last job's completion time}.\\[0.5em]
\emph{Ensures no job finishes too late.}
\end{block}

\column{0.48\textwidth}
\begin{block}{Total Completion Time}
\[
\min \; \sum_{j \in J} C_j
\]
The \textbf{sum of all completion times}.\\[0.5em]
\emph{Maximizes global throughput.}
\end{block}
\end{columns}

~\\

These two objectives are \textbf{conflicting}: improving one typically worsens the other.

\end{frame}

\section{Proposed Method}

\begin{frame}[plain]
\begin{center}
{\Huge Proposed Method:\\
\textsc{NS-BRKGA}}
\end{center}
\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Why NS-BRKGA?}

\textsc{NS-BRKGA} hybridizes two powerful frameworks:

~\\

\begin{columns}[c,onlytextwidth]
\column{0.48\textwidth}
\textbf{From \textsc{BRKGA-MP-IPR}:}
\begin{itemize}
\item Random-key chromosome representation
\item Multi-parent biased crossover
\item Implicit path-relinking
\item Shaking \& partial reset
\end{itemize}

\column{0.48\textwidth}
\textbf{From \textsc{NSGA-II}:}
\begin{itemize}
\item Non-dominated sorting
\item Crowding distance diversity
\item Pareto-based elite selection
\end{itemize}
\end{columns}

~\\

\textbf{Key advantage:} the evolutionary engine is \textbf{problem-independent}.
All MOTMDSP-specific logic resides in a \textbf{specialized decoder}.

\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Random-Key Representation}

A chromosome is a vector of real-valued keys:
\[
\mathbf{v} = (v_1, v_2, \ldots, v_{|J|}), \qquad v_j \in [0,\,1).
\]

~\\

\textbf{Decoding:} sort jobs in \textbf{decreasing order} of their keys
$\Rightarrow$ this defines a \textbf{priority order} for scheduling.

~\\

\begin{center}
\begin{tikzpicture}[>=Stealth, scale=0.9]
% Chromosome
\node[font=\footnotesize] at (-1.3,0) {$\mathbf{v}$:};
\foreach \val [count=\i from 0] in {0.81, 0.23, 0.65, 0.42, 0.97} {
    \node[draw, minimum width=1.1cm, minimum height=0.7cm, font=\footnotesize]
        at (\i*1.2, 0) {\val};
    \node[font=\scriptsize, above] at (\i*1.2, 0.45) {$j_{\the\numexpr\i+1}$};
}

% Arrow
\draw[-Stealth, ultra thick, mydarkorange1] (5.2,0) -- (6.2,0);

% Priority
\node[font=\footnotesize] at (6.6,0.45) {\textbf{Priority:}};
\foreach \job [count=\i from 0] in {$j_5$, $j_1$, $j_3$, $j_4$, $j_2$} {
    \node[draw, fill=mylightorange1!50, minimum width=0.9cm, minimum height=0.7cm, font=\footnotesize]
        at (7.7+\i*1.0, 0) {\job};
}
\end{tikzpicture}
\end{center}

~\\

Small perturbations in keys $\Rightarrow$ small changes in job order.
Enables \textbf{smooth exploration} of scheduling permutations.

\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Specialized Decoder}

Given a chromosome $\mathbf{v}$, the decoder builds a \textbf{feasible left-shifted schedule}:

~\\

\begin{enumerate}
\item \textbf{Sort} jobs by decreasing key value (priority order).

~\\

\item For each job in priority order:
    \begin{itemize}
    \item Assign to the \textbf{earliest feasible machine-period} intervals.
    \item Respect all constraints ($c_{mt}$, $r_{\max}$, $k_{\max}$, $r_{\min}$).
    \item Continue until demand $d_j$ is fully served.
    \end{itemize}

~\\

\item Compute $C_j$ for each job, then output $\mathbf{F} = (C_{\max},\; \sum_j C_j)$.
\end{enumerate}

~\\

The decoder \textbf{always produces a feasible schedule}: no repair or penalty mechanism is needed.

\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Additional NS-BRKGA Features}

\begin{columns}[c,onlytextwidth]
\column{0.50\textwidth}
\begin{itemize}
\item \textbf{Biased multi-parent crossover}\\
$\pi = 3$ parents, bias towards elite.

~\\

\item \textbf{Polynomial mutation}\\
Small perturbations of random keys.

~\\

\item \textbf{Genotypic diversity}\\
at elite selection step.

~\\

\item \textbf{Multiple populations}\\
with periodic migration.
\end{itemize}

\column{0.50\textwidth}
\begin{itemize}
\item \textbf{Implicit Path-Relinking}\\
Interpolates keys between elite solutions.

~\\

\item \textbf{Shaking}\\
Strong mutation on stagnation.

~\\

\item \textbf{Partial reset}\\
Injects fresh random individuals.
\end{itemize}
\end{columns}

\end{frame}

\section{Computational Experiments}

\begin{frame}[plain]
\begin{center}
{\Huge Computational\\Experiments}
\end{center}
\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Experimental Setup}

\begin{itemize}
\item \textbf{96 instances} from the realistic FOTA/TMDSP benchmark:
    \begin{itemize}
    \item \texttt{common}: vehicles appear in many sectors/periods.
    \item \texttt{rare}: vehicles appear in few sectors/periods.
    \item Up to \textbf{10,000 jobs} or \textbf{10,000 machines}.
    \end{itemize}

~\\

\item \textbf{5 baseline algorithms} (via PaGMO2, default parameters):\\
\textsc{NSGA-II}, \textsc{NSPSO}, \textsc{MOEA/D-DE}, \textsc{MHACO}, \textsc{IHS}.

~\\

\item \textbf{5 independent runs} per algorithm and instance.

~\\

\item \textbf{Time budget:} 900 seconds (15 min) per run.

~\\

\item CPU: Intel Xeon E3-1230 V2 @ 3.30\,GHz, 32\,GB RAM, single core.
\end{itemize}

\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Performance Indicators}

Two complementary quality indicators:

~\\

\begin{enumerate}
\item \textbf{Normalized Modified Inverted Generational Distance} ($\NIGDplus$)
\begin{itemize}
\item Measures \textbf{convergence} to the reference front.
\item \textbf{Lower is better}.
\end{itemize}

~\\

\item \textbf{Hypervolume Ratio} ($\HVR$)
\begin{itemize}
\item Measures both \textbf{convergence and diversity}.
\item \textbf{Higher is better}.
\end{itemize}
\end{enumerate}

~\\

Reference fronts: aggregation of all non-dominated solutions from \textbf{all algorithms across all runs}.

\end{frame}

\frametitle{Key Observations: \texttt{common} vs.\ \texttt{rare}}

\item All methods achieve nearly identical performance.
\item $\HVR \approx 1$ and $\NIGDplus \sim 10^{-5}$ for all algorithms.

\textbf{\texttt{rare} instances:}
\begin{itemize}
\item Convergence is \textbf{more challenging} for all algorithms.
\item \textsc{NS-BRKGA} stands out with:
    \begin{itemize}
    \item Best $\HVR$ ($0.99997$).
    \item Substantially lower $\NIGDplus$ ($0.00003$ vs.\ $0.00008$--$0.00016$).
    \end{itemize}
\item Indicates greater \textbf{robustness} under irregular availability patterns.
\end{itemize}

\frametitle{Conclusion}

\begin{itemize}
\item Proposed the \textbf{MOTMDSP}: a bi-objective extension of TMDSP that captures the trade-off between makespan and total completion time.

~\\

\item Developed an \textbf{\textsc{NS-BRKGA}} approach with a specialized greedy decoder.

~\\

\item Extensive experiments on \textbf{96 large-scale FOTA instances} show:\\
\textsc{NS-BRKGA} is consistently competitive or best on both $\HVR$ and $\NIGDplus$.

~\\

\item Especially strong on \textbf{\texttt{rare} instances}: greater robustness under atypical availability.

~\\

\item \textbf{Practical impact:} diverse Pareto-optimal schedules that help decision makers balance throughput vs.\ last completion time.
\end{itemize}
